
### tmux組織間通信プロトコル
```bash
# 実証済み通信パターン
tmux send-keys -t [pane] "[message]"  # メッセージ送信
tmux send-keys -t [pane] Enter        # Enter別送信（必須）
sleep 3                               # 受信確認待機
tmux capture-pane -t [pane] -p        # 応答確認

# 標準タスク指示フォーマット
TASK_INSTRUCTION_FORMAT=(
    "依頼元：pane-X: 役割名"
    "依頼先：pane-Y: 役割名"
    "タスク種別：組織実行"
    "依頼内容：（具体的詳細）"
    "報告：完了報告義務・フォーマット"
)
```

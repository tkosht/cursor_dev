    local task_type="$1"
    local task_name="$2"
    local complexity="$3"
    local timeline="$4"
    
    echo "ğŸ—ï¸ Generating custom checklist"

            ;;

            local base_template="bug_fix_checklist.md"

            ;;

            local base_template="project_management_checklist.md"
            ;;

            echo "âŒ Unknown task type: $task_type"
            return 1
            ;;

    # Complexity adjustment
    case "$complexity" in
        "simple")
            reduce_checklist_items "$base_template" 30

            # Use template as-is
